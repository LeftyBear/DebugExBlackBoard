Private Sub ListView長期休業日_DblClick()
    Dim TargetListView As MSComctlLib.ListView
    Set TargetListView = MeListView("ListView長期休業日")
    Dim ColumnsCount As Long
    ColumnsCount = TargetListView.ColumnHeaders.Count
    With FormForListView
        Dim TargetLabels() As MSForms.Control
        ReDim TargetLabels(1 To ColumnsCount)
        Dim i As Long
        For i = 1 To ColumnsCount
            Set TargetLabels(i) = .Controls.Add("Forms.Label.1", TargetListView.ColumnHeaders.Item(i).Text)
            With TargetLabels(i)
                .Top = 10
                .Height = 20
                If i = 1 Then
                    .Left = 20
                Else
                    .Left = TargetLabels(i - 1).Left + TargetLabels(i - 1).Width + 20
                End If
                .Width = TargetListView.ColumnHeaders.Item(i).Width
                Dim FormWidth As Single
                FormWidth = FormWidth + .Width
                Dim TargetLabel As MSForms.Label
                Set TargetLabel = TargetLabels(i)
                TargetLabel.Caption = TargetLabels(i).Name
                TargetLabel.Font.Name = "UD デジタル 教科書体 N-R"
            End With
        Next
        Dim TargetTextBoxes() As MSForms.Control
        ReDim TargetTextBoxes(1 To 1)
        Set TargetTextBoxes(1) = .Controls.Add("Forms.TextBox.1")
        With TargetTextBoxes(1)
            .Top = TargetLabels(1).Top + TargetLabels(1).Height
            .Height = 20
            .Left = TargetLabels(1).Left
            .Width = TargetLabels(1).Width
            Dim TargetTextBox As MSForms.TextBox
            Set TargetTextBox = TargetTextBoxes(1)
            TargetTextBox.SpecialEffect = fmSpecialEffectFlat
            TargetTextBox.BorderStyle = fmBorderStyleSingle
            TargetTextBox.Font.Name = "UD デジタル 教科書体 N-R"
        End With
        Dim TargetComboBoxes() As MSForms.Control
        ReDim TargetComboBoxes(1 To 2)
        For i = 1 To 2
            Set TargetComboBoxes(i) = .Controls.Add("Forms.ComboBox.1")
            With TargetComboBoxes(i)
                .Top = TargetLabels(i + 1).Top + TargetLabels(i + 1).Height
                .Height = 20
                .Left = TargetLabels(i + 1).Left
                .Width = TargetLabels(i + 1).Width
                Dim TargetComboBox As MSForms.ComboBox
                Set TargetComboBox = TargetComboBoxes(i)
                TargetComboBox.SpecialEffect = fmSpecialEffectFlat
                TargetComboBox.BorderStyle = fmBorderStyleSingle
                TargetComboBox.Font.Name = "UD デジタル 教科書体 N-R"
            End With
        Next
        Dim TargetCommandButtons(1) As MSForms.Control
        Set TargetCommandButtons(1) = .Controls.Add("Forms.CommandButton.1")
        With TargetCommandButtons(1)
            .Top = TargetComboBoxes(2).Top + TargetComboBoxes(2).Height + 10
            .Height = 25
            .Left = TargetComboBoxes(2).Left
            .Width = TargetComboBoxes(2).Width
            Dim TargetCommandButton As MSForms.CommandButton
            Set TargetCommandButton = TargetCommandButtons(1)
            TargetCommandButton.Caption = "登録"
            TargetCommandButton.Font.Name = "UD デジタル 教科書体 N-R"
            Set FormForListViewButton = TargetCommandButton
        End With
        .Width = FormWidth + i * 20 + 30
        .Caption = "長期休業日"
    End With
    Dim CalenderForComboBox As ICalenderForComboBox
    Set CalenderForComboBox = New CalenderForComboBox
    With CalenderForComboBox
        .Add FormForListView, TargetComboBoxes(1), 1
        .Add FormForListView, TargetComboBoxes(2), 2
    End With
    FormForListView.Show vbModal
End Sub
Private Sub FormForListViewButton_Click()
    MsgBox "OK"
End Sub
